=== 6.`Array.forEach()`

[playground]
const nums = [1, 51, 22, "b", 57, 70, 10, 2, "a"];
const numUnder50 = [];
const numOver50 = [];
const notNum = [];
// ----
nums.forEach(function (i) {
  if (typeof i !== "number") {
    notNum.push(i);
  } else if (i > 50) {
    numOver50.push(i);
  } else {
    numUnder50.push(i);
  }
});
console.log(notNum); // [ 'b', 'a' ]
console.log(numOver50); // [ 51, 57, 70 ]
console.log(numUnder50); // [ 1, 22, 10, 2 ]
